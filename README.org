** hello
#+BEGIN_SRC
  /* ~/hello/main.kl */
  import "io"
  func main() {
    io.print("hello, world")
  }
#+END_SRC
** compile
#+BEGIN_SRC
  :~$ cd ~/
  :~$ koalac hello
#+END_SRC
** run
#+BEGIN_SRC
  :~$ cd ~/
  :~$ koala hello
#+END_SRC
